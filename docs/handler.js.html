const axios = require('axios')
const version = require('../package').version
const {BASE_URL, ENDPOINTS, FILE_TYPES, DISCORD_STATUSES} = require('./base')

/**
 * Main Handler for Weeb.sh
 */
class WeebHandler {
  /**
   * @class WeebHandler - Main handler for all Weeb.sh request
   * @param {String} key - The token to use for Requesting all data prefixed with the Token type.
   * @param {String} [agent=Weeb.js/{version}] - The User Agent to attach to all requests (Formatted as NAME/VERSION or NAME/VERSION/ENV)
   */
  constructor (key, agent = 'Weeb.js/' + version) {
    if (typeof key !== 'string') throw new TypeError('Authorization Key is not a string.')
    let type = 'Wolke'
    if (key.startsWith('Bearer')) type = 'Bearer'
    const toSlice = key.startsWith('Bearer') || key.startsWith('Wolke') ? type.length : 0
    if (typeof agent !== 'string') throw new TypeError('User-Agent is not a string.')

    /**
     * User Agent that is being attached to all requests
     * @type {string}
     */
    this.agent = agent
    /**
     * Authorization Key used for all requests
     * @type {string}
     */
    this.key = type + ' ' + key.slice(toSlice).trim()

    /**
     * Axios Client to make all request to Weeb.sj
     * @type {axios}
     */
    this.axios = axios.create({baseURL: BASE_URL, headers: {'Authorization': this.key, 'User-Agent': this.agent}})
  }

  /**
   * Get a list of all available Tags
   * @param {Object} [options={}] - Options for this request
   * @param {Boolean} [options.nsfw=false] - Whenever to include NSFW tags or not, can be true, false or only
   * @param {Boolean} [options.hidden=false] - Whenever to include hidden tags or not
   * @return {Promise<WeebTagsResponse>}
   */
  getTags (options = {}) {
    if (options.hidden && typeof options.hidden !== 'boolean') throw new TypeError('Hidden must be a boolean')
    if (options.nsfw && typeof options.nsfw !== 'boolean' && options.nsfw !== 'only') throw new Error('NSFW must either be true, false or only')
    const query = {nsfw: options.nsfw || false, hidden: options.hidden || false}

    return this.requestAPI('tags', query)
  }

  /**
   * Get a list of all available Types
   * @param {Object} [options={}] - Options for this request
   * @param {Boolean} [options.nsfw=false] - Whenever to include NSFW types or not, can be true, false or only
   * @param {Boolean} [options.hidden=false] - Whenever to include hidden types or not
   * @param {Boolean} [options.preview=false] - Whenever to include a preview for each type or not
   * @return {Promise<WeebTypesResponse>}
   */
  getTypes (options = {}) {
    if (options.hidden && typeof options.hidden !== 'boolean') throw new TypeError('Hidden must be a boolean')
    if (options.preview && typeof options.preview !== 'boolean') throw new TypeError('Preview must be a boolean')
    if (options.nsfw && typeof options.nsfw !== 'boolean' && options.nsfw !== 'only') throw new Error('NSFW must either be true, false or only')

    const query = {nsfw: options.nsfw || false, hidden: options.hidden || false}
    // Preview just needs to be present to work even setting it to false would return previews
    if (options.preview) query['preview'] = true
    return this.requestAPI('types', query)
  }

  /**
   * Retrieve a random image using either a type or tags
   * @param {Object} [options={}] - Options for this request
   * @param options
   * @param {Boolean} [options.nsfw=false] - Whenever to include NSFW images or not, can be true, false or only
   * @param {Boolean} [options.hidden=false] - Whenever to include hidden images or not
   * @param {String} [options.type=null] - The type category the image should belong to
   * @param {Array} [options.tags=[]] - Tags the image should have
   * @param {String} [options.filetype=null] - The file type the image returned can have (png, jpg, jpeg, gif), null means all.
   * @return {Promise<WeebImageReponse>}
   */
  getRandom (options = {}) {
    if (!options.type && !options.tags) throw new Error('Either tags or type are required.')
    if (options.tags && !(options.tags instanceof Array)) throw new TypeError('Tags is not an array')
    if (options.nsfw && typeof options.nsfw !== 'boolean' && options.nsfw !== 'only') throw new Error('NSFW must either be true, false or only')
    if (options.hidden && typeof options.hidden !== 'boolean') throw new TypeError('Hidden must be a boolean')
    if (options.filetype && typeof options.filetype !== 'string') throw new TypeError('Filetype must be a string')
    if (options.filetype && !FILE_TYPES.includes(options.filetype.toLowerCase())) throw new Error('Filetype must be either jpeg, jpg, png or gif')

    const query = {
      hidden: options.hidden || false,
      nsfw: options.nsfw || false,
      filetype: options.filetype || null,
      tags: options.tags || null,
      type: options.type || null
    }
    return this.requestAPI('random', query)
  }

  /**
   * Generate a Discord Status mock up using a given Image
   * @param {String} [statusIcon=online] - The status to mock, can be online, dnd, offline, streaming and idle
   * @param {String} url - The Image to perform the Mock on
   * @return {Promise<Buffer|Object>}
   */
  generateDiscordMock (url, statusIcon = null) {
    if (statusIcon && !DISCORD_STATUSES.includes(statusIcon.toLowerCase())) throw new Error('Status can only be online, dnd, idle, offline or streaming.')
    if (!url.match(/^https?:\/\/.+\.(?:jpg|png|jpeg|gif|webp)$/g)) throw new Error('You need to prove a valid direct image URL.')

    const query = {status: statusIcon || 'online', avatar: encodeURIComponent(url) || ''}
    return this.requestAPI('discord_status', query)
  }

  /**
   * Generate a simple image using Weeb.sh. Requires the generate_simple Scope!
   * @param {String} type - The type of Simple image to generate, can be awooo, eyes and won
   * @param {Object} [options={}] - Options for this request (only used when type is awooo)
   * @param {String} [options.face=fff0d3] - The hex code to use for the Face color
   * @param {String} [options.hair=cc817c] - The hex code to use for the hair color
   * @return {Promise<Buffer>}
   */
  generateImage (type, options = {}) {
    if (typeof type !== 'string' || (type !== 'awooo' && type !== 'eyes' && type !== 'won')) throw new TypeError('Type must be a string and either awooo, eyes or won')

    if (type === 'awoo' && (options.hair || options.face)) {
      if (options.hair && !options.hair.match(/^[0-9a-f]{6}$/gi)) throw new Error('Hair must be a valid hex code')
      if (options.face && !options.face.match(/^[0-9a-f]{6}$/gi)) throw new Error('Face must be a valid hex code')
    }
    const query = {type, face: options.face || 'fff0d3', hair: options.hair || 'cc817c'}
    return this.requestAPI('generate', query)
  }

  /**
   * Generate a License using Weeb.sh. Requires the generate_license Scope!
   *  @param {Object} [options={}] - Options for this request
   *  @param {String} options.title - The title of the license
   *  @param {String} options.icon - The main icon for the License (needs to be a valid URL)
   *  @param {Array} [options.badges=[]} - Badges to display below the main Image (all of these need to be valid URLs)
   *  @param {Array} [options.widgets=[]} - Widget text to display right of the main icon
   * @return {Promise<Buffer>}
   */
  generateLicense (options = {}) {
    if (typeof options.avatar !== 'string' || typeof options.title !== 'string') throw new TypeError('Both Title and Icon need to be given and must be a string.')
    if (options.badges && (!(options.badges instanceof Array) || options.badges.length > 3 || options.badges.length < 1)) throw new TypeError('Badges must be an array that contains at least 1 and not more than 3 values')
    if (options.widgets && (!(options.widgets instanceof Array) || options.widgets.length > 3 || options.widgets.length < 1)) throw new TypeError('Widgets must be an array that contains at least 1 and not more than 3 values')

    return this.requestAPI('license', {
      title: options.title,
      avatar: options.avatar,
      widgets: options.widgets || [],
      badges: options.badges || []
    })
  }

  /**
   * Generate a Waifu insult using a given Image. Requires the generate_waifu_insult Scope!
   * @param {string} URL - The Image to waifu insult
   * @return {Promise<Buffer>}
   */
  generateInsult (URL) {
    if (typeof URL !== 'string') throw new TypeError('The URL must be a string')

    return this.requestAPI('waifu_insult', {avatar: URL}, true)
  }

  /**
   * Generate a Love Ship using 2 Images. Requires the generate_love_ship Scope!
   * @param {String} targetOne - URL pointing to the first Image
   * @param {String} targetTwo - URL pointing to the second Image
   * @return {Promise<Buffer>}
   */
  generateLoveShip (targetOne, targetTwo) {
    if (typeof targetOne !== 'string' || typeof targetTwo !== 'string') throw new TypeError('Both targetOne and targetTwo need to be given and need to be a string')

    return this.requestAPI('love_ship', {targetOne, targetTwo}, true)
  }

  /**
   * Send a request to Weeb.sh
   * @param endpoint
   * @param query
   * @param isPost
   * @return {Promise<Buffer|Object>}
   */
  requestAPI (endpoint, query = {}, isPost = false) {
    return new Promise(async (resolve, reject) => {
      try {
        let res
        if (isPost) {
          res = await this.axios.post(ENDPOINTS[endpoint], query, {
            headers: {'Content-Type': 'application/json'},
            responseType: 'arraybuffer'
          })
        } else {
          res = await this.axios.get(ENDPOINTS[endpoint], {params: query})
        }
        let {data, status} = res
        if (status < 200 || status > 300) reject(Buffer.isBuffer(data) ? data.toString() : data)
        else resolve(data)
      } catch (err) {
        reject(err)
      }
    })
  }
}

module.exports = WeebHandler
